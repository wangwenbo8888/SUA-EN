 5function tissue_params = set_tissue_params
 6
 7%   Tissue density: kg/m^3
 8rou = 1138;
 9
10%   Frenquency: Hz
11freq_megahertz = 1.36;
12freq = freq_megahertz * 1e6;
13
14%   Velocity: m/s
15c = 1569;
16
17%   Attenuation coefficient: Np/m/MHz
18atten = 9;
19alpha = atten * freq_megahertz;
20
21%   Wavenumber
22K = 2*pi*freq/c;
23
24tissue_params = struct('density', rou,...
25                       'frequency', freq,...
26                       'velocity', c,...
27                       'attenuation', alpha,...
28                       'wavenumber', K);
