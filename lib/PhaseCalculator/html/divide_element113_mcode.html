 4function [element_coords, delta_area] = divide_element(...
 5    element_radius, count_in_radius, count_in_angle)
 6
 7radii = element_radius/count_in_radius/2 :...
 8        element_radius/count_in_radius :...
 9        element_radius - element_radius/count_in_radius/2;
10angles = 2*pi/count_in_angle : 2*pi/count_in_angle : 2*pi;
11
12element_coords_x = radii' * cos(angles);
13element_coords_y = radii' * sin(angles);
14element_coords_z = zeros(count_in_radius, count_in_angle);
15
16delta_area = zeros(size(element_coords_x));
17
18angle_index = 0;
19% comment when generating the C codes
20% for angle = angles
21for angle = 2*pi/count_in_angle : 2*pi/count_in_angle : 2*pi
22    angle_index = angle_index + 1;
23    radius_index = 0;
24    % comment when generating the C codes
25%     for radius = radii
26    for radius = element_radius/count_in_radius/2 :...
27        element_radius/count_in_radius :...
28        element_radius - element_radius/count_in_radius/2
29        radius_index = radius_index + 1;
30        if radius_index == 1
31            delta_area(radius_index, angle_index) = radius.^2*pi/count_in_angle;
32        else
33            delta_area(radius_index, angle_index) = ...
34                radius.^2*pi/count_in_angle - sum(delta_area(:,angle_index));
35        end
36    end
37end
38
39delta_area = delta_area(:);
40element_coords_x = element_coords_x(:);
41element_coords_y = element_coords_y(:);
42element_coords_z = element_coords_z(:);
43element_coords = [element_coords_x element_coords_y element_coords_z];
