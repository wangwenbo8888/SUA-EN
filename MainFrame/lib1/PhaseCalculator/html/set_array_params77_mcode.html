 4function array_params = set_array_params
 5
 6%   Radius of curvature: m
 7roc = 0.140;
 8
 9%   Radius of element: m
10element_radius = 0.005;
11
12%   Divide an element in the directions of raidus and angle.
13count_in_radius = 20;
14count_in_angle = 20;
15
16%   Ring number in the array
17ring_count = 6;
18
19%   Element number in each ring of the array
20element_in_ring_count = 24 * ones(1,ring_count);
21
22% The element number in the array
23element_count = sum(element_in_ring_count);
24
25%   The angle between the ring and the propagation axis: rad
26angle_with_propa = [23.58 26.67 29.40 31.72 34.29 36.76] / 180 * pi;
27
28%   The angle of the first element in the ring in the plane perpendicular
29%   with propagation: rad
30angle_vertical_propa = [0 7.25 0 7.25 0 7.25] / 180 *pi;
31
32%   
33angles_with_propa = [];
34angles_vertical_propa = [];
35
36for i = 1:ring_count
37    angles_with_propa = [angles_with_propa...
38        angle_with_propa(i) * ones(1, element_in_ring_count(i))];
39    angles_vertical_propa = [angles_vertical_propa...
40        angle_vertical_propa(i):...
41        2*pi/element_in_ring_count(i):...
42        angle_vertical_propa(i) + (element_in_ring_count(i) - 1) * ...
43        2*pi/element_in_ring_count(i)];
44end
45
46array_params = struct('roc', roc,...
47                      'radius', element_radius,...
48                      'count_in_radius', count_in_radius,...
49                      'count_in_angle', count_in_angle,...
50                      'ring_count', ring_count,...
51                      'element_in_ring_count', element_in_ring_count,...
52                      'element_count', element_count,...
53                      'angle_with_propa', angle_with_propa,...
54                      'angle_vertical_propa', angle_vertical_propa,...
55                      'angles_with_propa', angles_with_propa,...
56                      'angles_vertical_propa', angles_vertical_propa);
