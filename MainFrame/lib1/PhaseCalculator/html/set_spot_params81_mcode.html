 4function spot_params = set_spot_params(focus_count, coords)
 5
 6%   Intensity at the focal spot: W/m^3
 7intensity = 3.33e5;
 8
 9tissue_params = set_tissue_params;
10
11rou = tissue_params.density;
12c = tissue_params.velocity;
13
14array_params = set_array_params;
15%   Coordinates of the focal spot -> vector has three elements (x, y, z): m
16coords(3) = array_params.roc - coords(3);
17
18%   Pressure at the focal spot: Pa
19pressure = sqrt(2*rou*c*intensity);
20
21spot_params = struct('focus_count', focus_count,...
22                     'coords', coords,...
23                     'intensity', intensity,...
24                     'pressure', pressure);
