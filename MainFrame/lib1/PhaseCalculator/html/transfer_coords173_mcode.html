 4function coords_cartesian = transfer_coords(center_coord, element_coords)
 5
 6center_coord = center_coord(:);
 7
 8center_x = center_coord(1);
 9center_y = center_coord(2);
10center_z = center_coord(3);
11
12%   Angle between z-axis and negative of center
13theta = acos(-center_z / sqrt(center_x^2 + center_y^2 + center_z^2));
14
15r = sqrt(center_x^2 + center_y^2);
16if r < 1e-6
17    phi = pi/2;
18else
19    phi = acos(-center_x/r);
20end
21if center_y > 0
22    phi = 2*pi - phi;
23end
24
25matrix_pre = [ 0 1 0;...
26              -1 0 0;...
27               0 0 1];
28matrix_rotation_y = [cos(theta) 0 sin(theta);...
29                       0        1   0;...
30                    -sin(theta) 0 cos(theta)];
31matrix_rotation_z = [cos(phi) -sin(phi) 0;...
32                     sin(phi)  cos(phi) 0;...
33                       0         0      1];
34                   
35coords = matrix_rotation_z * matrix_rotation_y * matrix_pre * element_coords';
36
37translation = repmat(center_coord, 1, size(coords, 2));
38coords_cartesian = coords + translation;
39coords_cartesian = coords_cartesian';
