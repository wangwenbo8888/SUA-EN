 4function H = forward_matrix(element_coords, delta_area, spot_params)
 5
 6array_params = set_array_params;
 7element_count = array_params.element_count;
 8roc = array_params.roc;
 9angles_with_propa = array_params.angles_with_propa;
10angles_vertical_propa = array_params.angles_vertical_propa;
11
12focus_count = spot_params.focus_count;
13focus_coords = spot_params.coords;
14coords = zeros(focus_count, 3);
15coords(1, :) = focus_coords;
16if focus_count == 2
17    coords(2, :) = -focus_coords(1, :);
18end
19
20tissue_params = set_tissue_params;
21rou = tissue_params.density;
22freq = tissue_params.frequency;
23alpha = tissue_params.attenuation;
24K = tissue_params.wavenumber;
25
26center_coords = zeros(size(angles_vertical_propa, 2), 3);
27center_coords(:, 1) = roc * sin(angles_with_propa(:)) .* ...
28                            cos(angles_vertical_propa(:));
29center_coords(:, 2) = roc * sin(angles_with_propa(:)) .* ...
30                            sin(angles_vertical_propa(:));
31center_coords(:, 3) = roc * cos(angles_with_propa(:));
32
33H = (1i * rou * freq) * ones(focus_count, element_count);
34for index=1:element_count
35    coords_cartesian = transfer_coords(center_coords(index, :), element_coords);
36    r = zeros(size(coords_cartesian, 1), focus_count);
37    for count=1:focus_count
38        r(:, count) = sqrt(sum((coords_cartesian -... 
39        repmat(coords(count, :), size(coords_cartesian, 1), 1)).^2, 2));
40    end
41    temp = exp(-(1i*K + alpha)*r) .* repmat(delta_area, 1, focus_count) ./ r;
42    H(:, index) = H(:, index) .* sum(temp, 1).'; 
43end
